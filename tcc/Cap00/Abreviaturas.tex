\chapter*{Lista de Símbolos}
\addcontentsline{toc}{chapter}{Lista de Símbolos}

\begin{description}
    \item[\(\mathbf{Q}\)] \quad Espaço de configuração do robô.
    
    \item[\(\mathbf{Q_{free}}\)] \quad Espaço de configuração livre de colisões.
    
    \item[\(\mathbf{Q_{obs}}\)] \quad Região do espaço de configuração ocupada por obstáculos.
    
    \item[\(\mathbf{q_{start}}\)] \quad Configuração inicial do robô.
    
    \item[\(\mathbf{q_{goal}}\)] \quad Configuração objetivo.
    
    \item[\(\mathbf{q_{rand}}\)] \quad Configuração amostrada aleatoriamente.
    
    \item[\(\mathbf{q_{near}}\)] \quad Nó mais próximo ou nó selecionado para expansão.
    
    \item[\(\mathbf{q_{new}}\)] \quad Nova configuração gerada pelo algoritmo.
    
    \item[\(\mathbf{T}\)] \quad Árvore de planejamento construída pelo algoritmo.
    
    \item[\(\boldsymbol{\alpha}\)] \quad Probabilidade de seleção da componente global do HybridEST.
    
    \item[\(\boldsymbol{\rho(q_i)}\)] \quad Estimativa de densidade local ao redor do nó \(q_i\).
    
    \item[\(\mathbf{r_d}\)] \quad Raio utilizado para estimar a densidade local.
    
    \item[\(\mathbf{J(c)}\)] \quad Função de custo associada a um caminho \(c\).
    
    \item[\(\mathbf{t_{plan}}\)] \quad Tempo de planejamento.
    
    \item[\(\mathbf{t_{exec}}\)] \quad Tempo de execução do caminho pelo robô.
    
    \item[\(\mathbf{e_{final}}\)] \quad Erro final entre a posição do robô e o objetivo.
    
    \item[\(\mathbf{c_{min}}\)] \quad Clearance mínima do caminho em relação aos obstáculos.
\end{description}
